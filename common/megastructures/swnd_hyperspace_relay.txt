swnd_hyperspace_relay = {
	entity = "hyperspace_beacon_entity"
	construction_entity = "hyperspace_beacon_entity"
	portrait = "GFX_megastructure_hyperspace_beacon"
	place_entity_on_planet_plane = no
	build_outside_gravity_well = yes
	show_galactic_map_icon = no
	show_in_outliner = no
	construction_blocks_and_blocked_by = self_type
	build_time = 360
	resources = {
		category = megastructures_hyper_relay
		cost = {
			alloys = 500
			electronics = 250
			influence = 25
		}

		upkeep = {
			energy = 2
		}
	}
	bypass_type = relay_bypass

	potential = {
		always = no
	}

	possible = {
		exists = starbase
		custom_tooltip = {
			fail_text = "requires_inside_border_or_subject"
			OR = {
				is_inside_border = from
				AND = {
					exists = owner
					owner = {
						is_subject = yes
						overlord = { is_same_value = from }
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_surveyed_system" # or is in my subject's territory
			OR = {
				NOT = {
					any_system_planet = {
						is_surveyed = {
							who = prev.from
							status = no
						}
					}
				}
				AND = {
					exists = owner
					owner = {
						is_subject = yes
						overlord = { is_same_value = from }
					}
				}
			}
		}
		custom_tooltip = {
			fail_text = "requires_no_existing_hyper_relay"
			NOR = {
				has_megastructure = swnd_hyperspace_relay
				has_megastructure = swnd_hyperroute_relay
			}
		}
		custom_tooltip = {
			fail_text = "requires_hyperlane"
			count_neighbor_system = {
				count > 0
			}
		}
	}

	placement_rules = {			# those would be ignored since the hyper relay is being "free-placed" between the inner and outer radius of the system
	}
	on_build_start = {}
	on_build_cancel = { remove_star_flag = "sw_hyper_relay_build" }
	on_build_complete = {
		set_star_flag = "sw_hyper_relay_build"
		fromfrom = { set_name = { key = "NAME_megastructure_hyper_relay" variable_string = "[Solar_System.GetName]" } }
	}

	# root = system
	# from = country
	ai_weight = {
		factor = 3

		modifier = {
			factor = 3
			is_capital_system = yes
		}

		modifier = {
			factor = 2
			OR = {
				any_neighbor_system = {
					has_megastructure = swnd_hyperspace_relay
				}
				any_neighbor_system = {
					has_megastructure = swnd_hyperroute_relay
				}
			}
		}

		modifier = {
			factor = 2
			count_system_colony = {
				count > 0
			}
		}
		
		#Don't bankrupt yourself, sillies
		modifier = {
			factor = 0
			FROM = {
				OR = {
					has_monthly_income = { resource = energy value < 20 }
					has_monthly_income = { resource = alloys value < 10 }
					has_monthly_income = { resource = electronics value < 5 }
					has_country_resource = { type = influence amount < 125 }
					has_country_resource = { type = alloys amount < 750 }
					has_country_resource = { type = electronics amount < 500 }
				}
			}
		}

		# non-gestalt empires will prefer building along trade routes, since those connect colonies to the capital
		# this may occasionally catch other empire's trade routes that clip our territory, but pirates will eventually appear there so it's fine
		modifier = {
			factor = 0.2
			from = {
				is_gestalt = no
			}
			is_capital_system = no
			has_trade_route = no
		}
	}
}
